                            if (date('D', strtotime($this->active_year . '-' . $this->active_month . '-' . $i)) == 'Tue') {

                            if ($i == $this->active_day) {

                                                                                                  $date = isset($_GET['date']) ? date('Y-m-d', strtotime($_GET['date'])) : '2023-05-11';

                                                                                                  <a href="?date=<?=date('Y-m-d', strtotime($date . ' +1 month'))?>">Next Month</a>
